package com.example.volumekeymapper;

import android.accessibilityservice.AccessibilityService;
import android.content.Intent;
import android.os.Handler;
import android.os.Looper;
import android.util.Log;
import android.view.KeyEvent;
import android.view.accessibility.AccessibilityEvent;

public class VolumeKeyAccessibilityService extends AccessibilityService {

    private static final String TAG = "VolumeKeyService";
    private static VolumeKeyAccessibilityService instance;
    private static boolean functionEnabled = false;

    private Handler handler = new Handler(Looper.getMainLooper());

    @Override
    public void onCreate() {
        super.onCreate();
        instance = this;
        Log.d(TAG, "Accessibility service created");
    }

    @Override
    public void onDestroy() {
        super.onDestroy();
        instance = null;
        Log.d(TAG, "Accessibility service destroyed");
    }

    @Override
    public void onAccessibilityEvent(AccessibilityEvent event) {
        // 不需要处理AccessibilityEvent，主要用于按键拦截
    }

    @Override
    public void onInterrupt() {
        Log.d(TAG, "Accessibility service interrupted");
    }

    @Override
    protected boolean onKeyEvent(KeyEvent event) {
        // 只处理音量+键的按下事件
        if (event.getKeyCode() == KeyEvent.KEYCODE_VOLUME_UP &&
                event.getAction() == KeyEvent.ACTION_DOWN &&
                functionEnabled) {

            Log.d(TAG, "Volume up key intercepted, performing double back");
            performDoubleBack();
            return true; // 拦截事件，不传递给系统
        }

        return super.onKeyEvent(event);
    }

    /**
     * 执行双击返回操作
     */
    private void performDoubleBack() {
        // 第一次返回
        performGlobalAction(GLOBAL_ACTION_BACK);

        // 延迟100ms后执行第二次返回
        handler.postDelayed(() -> {
            performGlobalAction(GLOBAL_ACTION_BACK);
            Log.d(TAG, "Double back action completed");
        }, 100);
    }

    /**
     * 启用/禁用功能
     */
    public static void setFunctionEnabled(boolean enabled) {
        functionEnabled = enabled;
        Log.d(TAG, "Function " + (enabled ? "enabled" : "disabled"));

        // 通知悬浮窗更新状态
        if (instance != null) {
            Intent intent = new Intent("com.example.volumekeymapper.FUNCTION_STATE_CHANGED");
            intent.putExtra("enabled", enabled);
            instance.sendBroadcast(intent);
        }
    }

    /**
     * 获取功能状态
     */
    public static boolean isFunctionEnabled() {
        return functionEnabled;
    }

    /**
     * 获取服务实例
     */
    public static VolumeKeyAccessibilityService getInstance() {
        return instance;
    }

    /**
     * 检查服务是否运行
     */
    public static boolean isServiceRunning() {
        return instance != null;
    }
}
